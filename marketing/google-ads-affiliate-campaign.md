## Tracking Setup

### Google Ads Conversion Tracking Code Snippet

Place this in the `<head>` of `affiliates.html` and all affiliate redirect/click pages:

```html
<!-- Google Ads Conversion Tracking - DevBotAI Affiliate Clicks -->
<script async src="https://www.googletagmanager.com/gtag/js?id=AW-XXXXXXXXX"></script>
<script>
  window.dataLayer = window.dataLayer || [];
  function gtag(){dataLayer.push(arguments);}
  gtag('js', new Date());
  gtag('config', 'AW-XXXXXXXXX');
</script>
```

**Affiliate Link Click Conversion Event** (fire when user clicks an affiliate link):

```javascript
// Trigger this on affiliate link click events
function trackAffiliateClick(affiliateName, estimatedValue) {
  gtag('event', 'conversion', {
    'send_to': 'AW-XXXXXXXXX/YYYYYYYYYY', // Replace with your conversion ID/label
    'value': estimatedValue || 50.00,
    'currency': 'USD',
    'transaction_id': Date.now() + '_' + affiliateName,
    'event_callback': function() {
      // Optional: redirect after tracking
    }
  });

  // Also push to dataLayer for GTM
  dataLayer.push({
    'event': 'affiliate_click',
    'affiliate_name': affiliateName,
    'affiliate_value': estimatedValue,
    'page_location': window.location.href
  });
}

// Attach to affiliate links - example implementation
document.querySelectorAll('.affiliate-link, [data-affiliate]').forEach(function(link) {
  link.addEventListener('click', function(e) {
    var affiliateName = this.dataset.affiliate || this.href;
    var commissionValue = parseFloat(this.dataset.commission) || 50;
    trackAffiliateClick(affiliateName, commissionValue);
  });
});
```

**Remarketing Audience Pixel** (fire on affiliates.html page load):

```javascript
// Tag affiliates.html visitors for remarketing
gtag('event', 'page_view', {
  'send_to': 'AW-XXXXXXXXX',
  'dynx_itemid': 'affiliates_page',
  'dynx_pagetype': 'affiliate_directory',
  'dynx_totalvalue': 0
});

// Create custom audience: "Viewed Affiliates Page"
gtag('set', 'user_properties', {
  'viewed_affiliates': 'true',
  'visit_date': new Date().toISOString().split('T')[0]
});
```
